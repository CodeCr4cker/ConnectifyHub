    <script>
        // Inspirational quotes array
        const quotes = [
            "Every ending is a new beginning waiting to unfold",
            "The future belongs to those who believe in the beauty of their dreams",
            "Success is not final, failure is not fatal: it is the courage to continue that counts",
            "Your journey doesn't end here; it transforms into something greater",
            "The best is yet to come",
            "Chase your dreams and never look back",
            "New adventures await beyond these walls",
            "You have the power to write your own story",
            "Believe in yourself and all that you are",
            "The world is yours to explore and conquer"
        ];

        // Generate random future date within next 30 days
        function generateRandomDate() {
            const today = new Date();
            const futureDate = new Date(today.getTime() + (Math.random() * 30 + 5) * 24 * 60 * 60 * 1000);
            
            const options = { 
                year: 'numeric', 
                month: 'long', 
                day: 'numeric'
            };
            
            return futureDate.toLocaleDateString('en-US', options);
        }

        // Generate random quote
        function getRandomQuote() {
            return quotes[Math.floor(Math.random() * quotes.length)];
        }

        // Update display when name changes
        document.getElementById('studentName').addEventListener('input', function() {
            const name = this.value.trim();
            const displayName = document.getElementById('displayName');
            
            if (name) {
                displayName.textContent = name;
                displayName.style.transform = 'scale(1.05)';
                setTimeout(() => {
                    displayName.style.transform = 'scale(1)';
                }, 200);
            } else {
                displayName.textContent = 'Enter Your Name';
            }
            
            // Generate new quote and date when name changes
            document.getElementById('inspirationalQuote').textContent = getRandomQuote();
            document.getElementById('eventDate').textContent = generateRandomDate();
        });

        // Initialize with random quote and date
        document.addEventListener('DOMContentLoaded', function() {
            document.getElementById('inspirationalQuote').textContent = getRandomQuote();
            document.getElementById('eventDate').textContent = generateRandomDate();
        });

        // Enhanced 3D card effect
        document.addEventListener('DOMContentLoaded', function() {
            const card = document.querySelector('.invitation-card');
            
            card.addEventListener('mousemove', function(e) {
                const rect = card.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;
                
                const centerX = rect.width / 2;
                const centerY = rect.height / 2;
                
                const rotateX = (y - centerY) / 15;
                const rotateY = (centerX - x) / 15;
                
                card.style.transform = `
                    perspective(1000px) 
                    rotateX(${rotateX}deg) 
                    rotateY(${rotateY}deg) 
                    scale3d(1.02, 1.02, 1.02)
                `;
                card.style.transition = 'transform 0.1s ease-out';
            });
            
            card.addEventListener('mouseleave', function() {
                card.style.transform = 'perspective(1000px) rotateX(0deg) rotateY(0deg) scale3d(1, 1, 1)';
                card.style.transition = 'transform 0.6s cubic-bezier(0.23, 1, 0.32, 1)';
            });
        });

        // Download functionality with enhanced canvas rendering
        function downloadCard() {
            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');
            
            // High resolution for crisp output
            const scale = 4;
            canvas.width = 450 * scale;
            canvas.height = 650 * scale;
            ctx.scale(scale, scale);
            
            // Dark gradient background
            const bgGradient = ctx.createLinearGradient(0, 0, 450, 650);
            bgGradient.addColorStop(0, '#1e1e2e');
            bgGradient.addColorStop(1, '#2a2a3e');
            ctx.fillStyle = bgGradient;
            ctx.fillRect(0, 0, 450, 650);
            
            // Overlay gradient
            const overlayGradient = ctx.createLinearGradient(0, 0, 450, 650);
            overlayGradient.addColorStop(0, 'rgba(124, 58, 237, 0.1)');
            overlayGradient.addColorStop(0.5, 'rgba(236, 72, 153, 0.1)');
            overlayGradient.addColorStop(1, 'rgba(6, 182, 212, 0.1)');
            ctx.fillStyle = overlayGradient;
            ctx.fillRect(0, 0, 450, 650);
            
            // College logo background
            const logoGradient = ctx.createLinearGradient(185, 70, 265, 150);
            logoGradient.addColorStop(0, '#7c3aed');
            logoGradient.addColorStop(1, '#ec4899');
            ctx.fillStyle = logoGradient;
            ctx.beginPath();
            ctx.arc(225, 110, 40, 0, 2 * Math.PI);
            ctx.fill();
            
            // College logo text
            ctx.fillStyle = '#ffffff';
            ctx.font = 'bold 24px Arial';
            ctx.textAlign = 'center';
            ctx.fillText('IC', 225, 118);
            
            // College name
            ctx.fillStyle = '#94a3b8';
            ctx.font = '600 12px Arial';
            ctx.fillText('INTERNATIONAL COLLEGE', 225, 175);
            
            // Farewell title
            ctx.fillStyle = '#ffffff';
            ctx.font = 'bold 32px Arial';
            ctx.fillText('Farewell', 225, 210);
            
            // Honoring label
            ctx.fillStyle = '#7c3aed';
            ctx.font = '600 12px Arial';
            ctx.fillText('HONORING', 225, 280);
            
            // Student name
            ctx.fillStyle = '#ffffff';
            ctx.font = 'bold 36px Arial';
            const studentName = document.getElementById('studentName').value || 'Student Name';
            ctx.fillText(studentName, 225, 320);
            
            // Event details background
            ctx.fillStyle = 'rgba(0, 0, 0, 0.4)';
            ctx.roundRect(50, 360, 350, 140, 16);
            ctx.fill();
            
            // Event details
            ctx.fillStyle = '#94a3b8';
            ctx.font = '600 11px Arial';
            ctx.textAlign = 'left';
            
            const details = [
                ['DATE', document.getElementById('eventDate').textContent],
                ['TIME', '6:00 PM - 9:00 PM'],
                ['VENUE', 'Grand Auditorium'],
                ['DRESS CODE', 'Formal']
            ];
            
            details.forEach((detail, index) => {
                const y = 385 + (index * 30);
                ctx.fillStyle = '#94a3b8';
                ctx.fillText(detail[0], 70, y);
                ctx.fillStyle = '#ffffff';
                ctx.textAlign = 'right';
                ctx.fillText(detail[1], 380, y);
                ctx.textAlign = 'left';
            });
            
            // Quote
            ctx.fillStyle = '#cbd5e0';
            ctx.font = 'italic 14px Arial';
            ctx.textAlign = 'center';
            
            const quote = document.getElementById('inspirationalQuote').textContent;
            const maxWidth = 320;
            const words = quote.split(' ');
            let line = '';
            let y = 560;
            
            // Word wrapping for quote
            for (let i = 0; i < words.length; i++) {
                const testLine = line + words[i] + ' ';
                const metrics = ctx.measureText(testLine);
                
                if (metrics.width > maxWidth && i > 0) {
                    ctx.fillText(`"${line.trim()}"`, 225, y);
                    line = words[i] + ' ';
                    y += 20;
                } else {
                    line = testLine;
                }
            }
            
            if (line.trim()) {
                const finalLine = words.length > 5 ? `"${line.trim()}"` : `"${quote}"`;
                ctx.fillText(finalLine, 225, y);
            }
            
            // Quote author
            ctx.fillStyle = '#7c3aed';
            ctx.font = '600 10px Arial';
            ctx.fillText('— INTERNATIONAL COLLEGE', 225, y + 25);
            
            // Download
            const link = document.createElement('a');
            const fileName = studentName.replace(/\s+/g, '-').toLowerCase();
            link.download = `farewell-invitation-${fileName}.png`;
            link.href = canvas.toDataURL('image/png', 1.0);
            link.click();
        }

        // Enhanced rounded rectangle function
        CanvasRenderingContext2D.prototype.roundRect = function (x, y, width, height, radius) {
            this.beginPath();
            this.moveTo(x + radius, y);
            this.arcTo(x + width, y, x + width, y + height, radius);
            this.arcTo(x + width, y + height, x, y + height, radius);
            this.arcTo(x, y + height, x, y, radius);
            this.arcTo(x, y, x + width, y, radius);
            this.closePath();
        };
    </script>
